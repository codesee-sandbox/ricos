<RicosEditorWithMobile
    content={createContent('Add Link, to your favorite website')}

    plugins={[

      }),
    ]}

## Usage

```jsx
/* Editor */
import { RicosEditor } from 'ricos-editor';
import { pluginLink } from 'wix-rich-content-plugin-link';
import 'ricos-editor/dist/styles.min.css';
import 'wix-rich-content-plugin-link/dist/styles.min.css';
<RicosEditor plugins={[pluginLink(LinkConfig)]} />;

/* Viewer */
import { RicosViewer } from 'ricos-viewer';
import { pluginLink } from 'wix-rich-content-plugin-link/dist/module.viewer';
import 'wix-rich-content-plugin-link/dist/styles.min.css';
<RicosViewer plugins={[pluginLink(LinkConfig)]} />;
```

## Link Config API

### `onClick`

Optional Link click handler

```jsx
onClick?: (event: React.MouseEvent<HTMLElement>, text: string) => void;
```

Default value:

```jsx
none;
```

### `Link Preview`

The Link Preview Plugin is implemented within the Link Plugin,
and is activated once the Link Plugin Settings object contains preview field.
For further information check the PreviewPlugin page.

Default value:

```jsx
none;
```

### Link To Section

Anchor option allows users to navigate to a certain section of the website



<div className="card" >
<RicosEditorWithMobile
    content={createContent(`Try it yourself, add new lines and use the link to section option. Make sure you link to the lines you adde`)}
    placeholder={'Type here!'}
    plugins={[
      pluginLink()
    ]}
  />
</div>

### `linkTypes`

Supported Links types

Default value:

```jsx
{ anchor: true }
```

### `Site URL`

Anchor option changes the sites URL

This way links to certain sections of a website can be shared using the URL
